<script>
    // --- НАЛАШТУВАННЯ TELEGRAM СЮДИ ---
    const BOT_TOKEN = 'ВСТАВ_СЮДИ_ТВІЙ_ТОКЕН';
    const CHAT_ID = 'ВСТАВ_СЮДИ_ТВІЙ_CHAT_ID';
    // ---------------------------------

    let cart = [];
    let total = 0;

    function addToCart(name, price) {
        cart.push({name, price});
        total += price;
        document.getElementById('cart-count').innerText = cart.length;
        document.getElementById('cart-total').innerText = total;
        
        const cartBar = document.querySelector('.cart-bar');
        cartBar.style.transform = 'scale(1.02)';
        setTimeout(() => cartBar.style.transform = 'scale(1)', 150);
    }

    function openModal() {
        if(cart.length === 0) {
            alert("Кошик порожній! Додай послугу 🔥");
            return;
        }
        document.getElementById('orderModal').style.display = 'flex';
        document.getElementById('order-summary').innerText = `Разом до сплати: ${total} ₴ за ${cart.length} товар(и).`;
    }

    function closeModal() {
        document.getElementById('orderModal').style.display = 'none';
    }

    function sendOrder() {
        const name = document.getElementById('clientName').value.trim();
        const phone = document.getElementById('clientPhone').value.trim();
        const btn = document.getElementById('btnSubmit');

        if(!name || !phone) {
            alert("Будь ласка, заповни всі поля!");
            return;
        }

        btn.disabled = true;
        btn.innerText = "Надсилаємо...";

        let itemsList = cart.map(item => `• ${item.name} (${item.price} ₴)`).join('\n');
        let message = `🚀 *Нове замовлення в Upgrade Zone!*\n\n👤 *Клієнт:* ${name}\n📞 *Телефон:* ${phone}\n\n📦 *Товари:*\n${itemsList}\n\n💰 *Загальна сума:* ${total} ₴`;
        
        const url = `https://telegram.org{BOT_TOKEN}/sendMessage`;
        
        fetch(url, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                chat_id: CHAT_ID,
                text: message,
                parse_mode: 'Markdown'
            })
        })
        .then(response => {
            if(response.ok) {
                alert("🔥 Дякуємо за замовлення в Upgrade Zone! Ми вже зв'язуємось з тобою.");
                cart = [];
                total = 0;
                document.getElementById('cart-count').innerText = '0';
                document.getElementById('cart-total').innerText = '0';
                document.getElementById('clientName').value = '';
                document.getElementById('clientPhone').value = '';
                closeModal();
            } else {
                alert("Упс, сталася помилка при відправці. Перевір налаштування бота.");
            }
        })
        .catch(err => {
            alert("Помилка мережі. Спробуй ще раз.");
        })
        .finally(() => {
            btn.disabled = false;
            btn.innerText = "Підтвердити";
        });
    }
</script>
